 33def normalise_label(label: str) -> str:
 34    """Clean redundant suffixes in label strings for stable deduplication."""
 35    # Simplify redundant 'observed' stacking
 36    pattern = re.compile(r"(observed(?:,\s*observed)+)")
 37    label = pattern.sub("observed", label)
 38
 39    # If somehow multiple parentheses appear, strip down
 40    while "(observed, observed" in label:
 41        label = label.replace("(observed, observed", "(observed")
 42
 43    # Remove trailing junk commas
 44    label = re.sub(r"\(observed(?:,\s*)*\)", "(observed)", label)
 45
 46    return label

 48def canonicalise_holiday_data(holiday_data: dict) -> dict:
 49    """
 50    Deduplicate identical label+colour entries for each date before merge phase.
 51
 52    Args:
 53        holiday_data (dict): Raw holiday data dictionary with potential redundant entries.
 54
 55    Returns:
 56        dict: Cleaned dictionary with duplicate entries removed.
 57    """
 58    cleaned_data = {}
 59
 60    for date_key, data in holiday_data.items():
 61        seen = set()
 62        deduped_entries = []
 63
 64        for entry in data.get("entries", []):
 65            # Build tuple key (label, colour)
 66            entry_key = (normalise_label(entry["label"]), entry["colour"])
 67            if entry_key not in seen:
 68                seen.add(entry_key)
 69                deduped_entries.append(entry)
 70
 71        cleaned_data[date_key] = {"entries": deduped_entries}
 72
 73    return cleaned_data

 76def smart_merge_au_holidays(holiday_data):
 77    """
 78    For each date, merge holidays with identical names, appending all region codes into the label.
 79
 80    Args:
 81        holiday_data (MultiColourHolidayDict): {date: {"entries": [{"label", "colour"}, ...]}}
 82
 83    Returns:
 84        MultiColourHolidayDict: With merged entries.
 85    """
 86    import re
 87    merged = {}
 88
 89    for date, data in holiday_data.items():
 90        name_to_regions = {}
 91        name_to_colour = {}
 92
 93        for entry in data["entries"]:
 94            # Parse out name and region from label, e.g. "Christmas Day (NSW)"
 95            m = re.match(r"^(.*) \(([^)]+)\)$", entry["label"])
 96            if m:
 97                name, region = m.group(1).strip(), m.group(2).strip()
 98            else:
 99                name, region = entry["label"], "National"
100            if name not in name_to_regions:
101                name_to_regions[name] = []
102            if region not in name_to_regions[name]:
103                name_to_regions[name].append(region)
104            # Use the first colour (can be improved if you want to blend)
105            name_to_colour.setdefault(name, entry["colour"])
106
107        merged_entries = []
108        for name, regions in name_to_regions.items():
109            merged_label = f"{name} ({', '.join(regions)})"
110            merged_entries.append({"label": merged_label, "colour": name_to_colour[name]})
111        merged[date] = {"entries": merged_entries}
112
113    return merged

117def merge_identical_holidays(holiday_data, merge_enabled=True, default_colour="black"):
118    """
119    Merge holiday entries for each date when multiple countries share an identical holiday name.
120
121    Fully harmonised: stable canonicalisation for merge keys, safe preservation of display labels,
122    and correct handling of observed suffixes.
123
124    Args:
125        holiday_data (dict): Dictionary in MultiColourHolidayDict format,
126                             {date: {"entries": [{"label": "Name (CC)", "colour": ...}, ...]}}
127        merge_enabled (bool): If False, does not alter input; if True, merges identical holidays.
128        default_colour (str): Colour to use for merged entries if not set per-country.
129
130    Returns:
131        dict: Holiday data with merged entries where applicable.
132    """
133    if not merge_enabled:
134        return holiday_data
135
136    merged_data = {}
137
138    for date, data in holiday_data.items():
139        name_to_countries = defaultdict(list)
140        name_to_colours = defaultdict(list)
141        canonical_to_display = {}
142
143        for entry in data.get("entries", []):
144            label = entry["label"]
145
146            # Default extraction: assume full label unless we parse country suffix
147            name = label.strip()
148            cc = None
149
150            # Extract (CC) suffix if present
151            if '(' in label and label.endswith(')'):
152                name_part, cc_part = label.rsplit('(', 1)
153                name = name_part.strip()
154                cc = cc_part[:-1].strip()
155
156                # Patch: observed suffix is NOT a country code
157                if cc.lower() == "observed":
158                    cc = None
159                    name = f"{name} (observed)"
160
161            # Build canonical key for internal merge
162            display_name = name  # preserve original name for output
163            canonical_name = canonicalise_label(display_name)
164
165            name_to_countries[canonical_name].append(cc)
166            name_to_colours[canonical_name].append(entry.get("colour", default_colour))
167
168            if canonical_name not in canonical_to_display:
169                canonical_to_display[canonical_name] = display_name
170
171        merged_entries = []
172
173        for canonical_name, countries in name_to_countries.items():
174            display_name = canonical_to_display[canonical_name]
175            unique_countries = [c for c in countries if c]
176
177            # Colour selection logic (stable, deterministic)
178            colour_counts = defaultdict(int)
179            for colour in name_to_colours[canonical_name]:
180                colour_counts[colour] += 1
181
182            max_count = max(colour_counts.values())
183            most_common = [colour for colour, count in colour_counts.items() if count == max_count]
184
185            # Handle fully identical no-country situations
186            if all(cc is None for cc in name_to_countries[canonical_name]):
187                if len(set(name_to_colours[canonical_name])) > 1:
188                    entry_colour = "black"
189                else:
190                    entry_colour = name_to_colours[canonical_name][0]
191            else:
192                for colour in name_to_colours[canonical_name]:
193                    if colour in most_common:
194                        entry_colour = colour
195                        break
196                else:
197                    entry_colour = default_colour
198
199            # Build final merged label using display name
200            if len(unique_countries) > 1:
201                merged_label = f"{display_name} ({', '.join(sorted(unique_countries))})"
202            elif unique_countries:
203                merged_label = f"{display_name} ({unique_countries[0]})"
204            else:
205                merged_label = display_name
206
207            merged_entries.append({"label": merged_label, "colour": entry_colour})
208
209        merged_data[date] = {"entries": merged_entries}
210
211    return merged_data

214def get_multi_country_holidays(settings) -> MultiColourHolidayDict:
215    """
216    Retrieve and combine holidays for multiple countries in a multi-entry format.
217
218    Args:
219        settings (dict): Settings with 'holidays'/'countries' and 'year'.
220
221    Returns:
222        MultiColourHolidayDict: {ISO date: {"entries": [ {"label", "colour"}, ... ]}}
223    """
224    countries = settings.get('include_country_list', {}).get('countries', [])
225    year = int(settings['year'])
226    holiday_data: MultiColourHolidayDict = {}
227
228    for cc in countries:
229        colour = COUNTRY_COLOURS.get(cc, 'grey')
230        hdays = myholidays.country_holidays(cc, years=year)
231        for date, name in hdays.items():
232            date_key = date.strftime("%Y-%m-%d") if hasattr(date, "strftime") else str(date)
233            entry = {"label": f"{name} ({cc})", "colour": colour}
234            if date_key in holiday_data:
235                if entry not in holiday_data[date_key]["entries"]:
236                    holiday_data[date_key]["entries"].append(entry)
237            else:
238                holiday_data[date_key] = {"entries": [entry]}
239
240    return holiday_data

242def get_combined_holidays(settings) -> MultiColourHolidayDict:
243    """
244    Return a unified holiday dictionary for the calendar engine.
245
246    Combining UK region-colour-coded holidays, Australian state/territory holidays,
247    and/or multi-country holidays as per config.
248
249    Args:
250        settings (dict): Project settings.
251
252    Returns:
253        MultiColourHolidayDict: Holidays ready for calendar rendering.
254    """
255    all_holidays = {}
256
257    # Add UK holidays if enabled
258    if settings.get('include_days', {}).get('uk_holidays', False):
259        uk_holidays = get_uk_combined_holidays(int(settings['year']))
260        all_holidays.update(uk_holidays)
261
262    # Only process Country List if Country List Flag is Set.
263    if settings.get("include_days", {}).get("country_list", False):
264        # Parse countries list robustly
265        icl = settings.get('include_country_list', {})
266        if isinstance(icl, dict):
267            country_list = icl.get('countries', []).copy()
268        else:
269            country_list = list(icl)  # fallback if flat list
270
271        # Special-case: Australia (call dedicated AU function and remove from country list)
272        if "AU" in country_list:
273            au_holidays = get_au_combined_holidays(int(settings['year']))
274            au_holidays = smart_merge_au_holidays(au_holidays)
275            for date, data in au_holidays.items():
276                if date in all_holidays:
277                    for entry in data["entries"]:
278                        if entry not in all_holidays[date]["entries"]:
279                            all_holidays[date]["entries"].append(entry)
280                else:
281                    all_holidays[date] = data
282            country_list.remove("AU")
283
284        # Add other multi-country holidays if any left
285        if country_list:
286            patched_settings = settings.copy()
287            # update the countries list in the correct structure
288            if isinstance(icl, dict):
289                patched_settings["include_country_list"]["countries"] = country_list
290            else:
291                patched_settings["include_country_list"] = country_list
292            mc_holidays = get_multi_country_holidays(patched_settings)
293            for date, data in mc_holidays.items():
294                if date in all_holidays:
295                    for entry in data["entries"]:
296                        if entry not in all_holidays[date]["entries"]:
297                            all_holidays[date]["entries"].append(entry)
298                else:
299                    all_holidays[date] = data
300
301    return all_holidays

304def get_au_combined_holidays(year: int) -> MultiColourHolidayDict:
305    """
306    Retrieve and combine Australian national and state/territory holidays for a given year.
307
308    Holidays are grouped by date in ISO format (YYYY-MM-DD), with each date containing
309    one or more entries. Overlapping holidays from different Australian regions are merged
310    into the same day but retain their individual labels and colours.
311
312    Region colours:
313        - "#AD9200"      = Australia (default/national)
314        - "darkblue"  = New South Wales (NSW)
315        - "crimson"   = Victoria (VIC)
316        - "teal"      = Queensland (QLD)
317        - "saddlebrown" = South Australia (SA)
318        - "green"     = Western Australia (WA)
319        - "purple"    = Tasmania (TAS)
320        - "deepskyblue" = Northern Territory (NT)
321        - "#D28800"    = Australian Capital Territory (ACT)
322
323    Args:
324        year (int): The year to generate holidays for.
325
326    Returns:
327        MultiColourHolidayDict: A dictionary of ISO date strings to lists of holiday entries.
328    """
329    # All state/territory codes (as used in `holidays`)
330    subdivisions = [None, "NSW", "VIC", "QLD", "SA", "WA", "TAS", "NT", "ACT"]
331
332    holiday_data: MultiColourHolidayDict = {}
333
334    def add_holiday(source, region_code):
335        """
336        Add holidays from a specific region/territory into the combined dictionary.
337
338        Args:
339            source (dict): Holidays dict from holidays.country_holidays().
340            region_code (str or None): Subdivision/state code.
341        """
342        colour = AUS_REGION_COLOURS.get(region_code, "grey")
343        for date, name in source.items():
344            date_key = date.strftime("%Y-%m-%d") if hasattr(date, "strftime") else str(date)
345            label = f"{name} ({region_code if region_code else 'National'})"
346            entry: HolidayLine = {"label": label, "colour": colour}
347            if date_key in holiday_data:
348                if entry not in holiday_data[date_key]["entries"]:
349                    holiday_data[date_key]["entries"].append(entry)
350            else:
351                holiday_data[date_key] = {"entries": [entry]}
352
353    # Loop through national (None) and all subdivisions
354    for subdiv in subdivisions:
355        region_holidays = myholidays.country_holidays("AU", years=year, subdiv=subdiv) if subdiv else myholidays.country_holidays("AU", years=year)
356        add_holiday(region_holidays, subdiv)
357
358    return holiday_data

410def nth_weekday_of_month(year, month, weekday, n):
411    """
412    Get the date of the nth weekday (0=Monday) of a given month/year.
413
414    E.g., second Sunday in May = nth_weekday_of_month(2026, 5, 6, 2)
415    """
416    count = 0
417    for day in range(1, 32):
418        try:
419            dt = datetime.date(year, month, day)
420        except ValueError:
421            break
422        if dt.weekday() == weekday:
423            count += 1
424            if count == n:
425                return dt
426    return None

428def last_weekday_of_month(year, month, weekday):
429    """Get the date of the last weekday (0=Monday) of a given month/year."""
430    last_day = calendar.monthrange(year, month)[1]
431    for day in range(last_day, 0, -1):
432        dt = datetime.date(year, month, day)
433        if dt.weekday() == weekday:
434            return dt
435    return None
